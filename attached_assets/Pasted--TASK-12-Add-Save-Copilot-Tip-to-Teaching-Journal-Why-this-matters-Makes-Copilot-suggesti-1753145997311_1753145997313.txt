def save_copilot():

        data = request.json
        tip = CopilotTip(
            grade=data.get('grade', ''),
            subject=data.get('subject', ''),
            topic=data.get('topic', ''),
            sel_focus=data.get('sel', ''),
            content=data.get('content', '')
        )
        db.session.add(tip)
        db.session.commit()
        return jsonify({"success": True})
    except Exception as e:
        return jsonify({"error": str(e)}), 500
